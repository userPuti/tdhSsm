关闭窗口
<input id="txt" type="text" />
<a onclick="doClose()">关闭窗口</a>
<script>
$(function(){
var value=window.parent.setVal();//调用父页面的方法
$('#txt').val(value);
});
function getVal(){
alert($('#txt').val());
}
function doClose(){
var index = parent.layer.getFrameIndex(window.name); //获取窗口索引
parent.layer.close(index);//关闭iframe页面
}
</script>
